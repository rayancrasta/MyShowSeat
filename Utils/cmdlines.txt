#Kafka start
/opt/homebrew/bin/zookeeper-server-start /opt/homebrew/etc/zookeeper/zoo.cfg
/opt/homebrew/bin/kafka-server-start /opt/homebrew/etc/kafka/server.properties

#Kafka topic:
kafka-topics --create --bootstrap-server localhost:9092 --replication-factor 1 --partitions 1 --topic reservation_requests


# check contents of kafka topic
kafka-console-consumer --bootstrap-server localhost:9092 --topic reservation_requests --from-beginning
